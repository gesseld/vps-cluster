### **0.1 Node Provisioning with Genuine Multi-Datacenter Anti-Affinity**

```bash
#!/bin/bash
set -euo pipefail

# CRITICAL FIX: Spread nodes across three Hetzner locations for fault tolerance
# nbg1 = Nuremberg, fsn1 = Falkenstein, hel1 = Helsinki
LOCATIONS=(nbg1 fsn1 hel1)

for i in 1 2 3; do
  LOC=${LOCATIONS[$((i-1))]}
  hcloud server create \
    --name cx-node-${i} \
    --type cx21 \
    --image ubuntu-22.04 \
    --location ${LOC} \
    --network ${NETWORK_ID} \
    --ssh-key ${SSH_KEY_ID} \
    --user-data-from-file cloud-init-node.yaml \
    --label stack=production \
    --label node-role=control-plane \
    --label topology.kubernetes.io/zone=${LOC}
done

# Create dedicated volumes with explicit node affinity
hcloud volume create --name vol-postgres --size 50 --server cx-node-1
hcloud volume create --name vol-minio --size 100 --server cx-node-2
hcloud volume create --name vol-data --size 50 --server cx-node-3

sleep 60

NODE1_IP=$(hcloud server ip cx-node-1)
NODE2_IP=$(hcloud server ip cx-node-2)
NODE3_IP=$(hcloud server ip cx-node-3)

echo "Nodes provisioned: ${NODE1_IP}, ${NODE2_IP}, ${NODE3_IP}"
```

kubectl get pvc -A

echo -e "\n🌐 Services:"
kubectl get svc -A

echo -e "\n✅ Critical Component Validation:"
for component in postgres temporal redis minio qdrant meilisearch kratos oathkeeper caddy; do
  phase=$(kubectl get pod -l app=${component} -o jsonpath='{.items[0].status.phase}' 2>/dev/null || echo "NotFound")
  if [ "$phase" = "Running" ]; then
    echo "✅ ${component} is running"
  else
    echo "❌ ${component} is NOT running — phase=${phase}"
  fi
done

echo -e "\n📈 Resource Usage:"

kubectl get cronjob -A | grep backup

echo -e "\n📊 VPA Recommendations:"

          annotations:
            summary: "Pod is crash looping"
            description: "Pod {{ \$labels.pod }} is restarting frequently"
